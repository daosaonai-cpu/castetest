C Programming Loops: A Crash Course

Your guide to iteration and control flow in C.

The `for` Loop

Syntax:

for (initialization; condition; increment/decrement) {
   // code to be executed
}

Example:

#include <stdio.h>

int main() {
   int i;
   
   // Loop three times
   for (i = 0; i < 3; i++) {
      printf("Hello C!\n");
   }
   
   return 0;
}

The `while` Loop

Syntax:

while (condition) {
   // code to be executed
}

Example:

#include <stdio.h>

int main() {
   int i = 0; // Initialization
   
   while (i < 3) { // Condition
      printf("Hello C!\n");
      i++; // Increment
   }
   
   return 0;
}

The `do...while` Loop

Syntax:

do {
   // code to be executed
} while (condition);

Example:

#include <stdio.h>

int main() {
   int n = 1;
   
   do {
      printf("This runs at least once!\n");
      n++;
   } while (n < 1); // Condition is false, but loop ran once
   
   return 0;
}

The `break` Statement

Example:

#include <stdio.h>

int main() {
   int i;
   
   for (i = 1; i <= 5; i++) {
      if (i == 3) {
         break; // Exit the loop when i is 3
      }
      printf("%d\n", i);
   }
   
   return 0;
}

The `continue` Statement

Example:

#include <stdio.h>

int main() {
   int i;
   
   for (i = 1; i <= 5; i++) {
      if (i == 3) {
         continue; // Skip printing 3
      }
      printf("%d\n", i);
   }
   
   return 0;
}
